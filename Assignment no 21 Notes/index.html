JavaScript ES6 Features Documentation

1. Closure

A function can access variables from its parent scope even after the parent function has executed.

function createCounter() {
    let count = 0; // Local variable
    return function () {
        count++;
        return count;
    };
}

const counter = createCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

2. Destructuring

Extract values from arrays or objects into variables easily.

// Array destructuring
const fruits = ["apple", "banana", "cherry"];
const [first, second] = fruits;
console.log(first); // apple
console.log(second); // banana

// Object destructuring
const user = { name: "Ali", age: 22 };
const { name, age } = user;
console.log(name); // Ali
console.log(age); // 22

3. Default Parameters

Set default values for function parameters when arguments are missing.

function greet(name = "Guest") {
    console.log(`Hello, ${name}!`);
}

greet(); // Hello, Guest!
greet("Ali"); // Hello, Ali!

4. Rest Parameter

Accept unlimited arguments as an array.

function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3)); // 6
console.log(sum(10, 20, 30, 40)); // 100

5. Spread Operator

Spread array or object values into new ones.

// Array example
const numbers = [1, 2, 3];
const newNumbers = [...numbers, 4, 5];
console.log(newNumbers); // [1, 2, 3, 4, 5]

// Object example
const user = { name: "Ali", age: 22 };
const updatedUser = { ...user, city: "Karachi" };
console.log(updatedUser); // { name: "Ali", age: 22, city: "Karachi" }

6. Enhanced Object Literals

Create objects with cleaner syntax and dynamic property names.

const name = "Ali";
const age = 22;

// Property shorthand
const user = { name, age };
console.log(user); // { name: "Ali", age: 22 }

// Method shorthand
const person = {
    sayHi() {
        console.log("Hi!");
    },
};
person.sayHi(); // Hi!

// Computed property names
const key = "role";
const employee = {
    [key]: "Developer",
};
console.log(employee); // { role: "Developer" }
